\IfPackageLoaded{hyperref}{%
    \hypersetup{%
        %% Opciones generales
        draft=false,              %
        final=true,               %
        debug=false,              %
        hypertexnames=true,       % Usa nombres amigables para los enlaces
        naturalnames=false,       % Usa nombres naturales para los enlaces

        urlcolor=UVBlue,          % Color de las URL

        bookmarksopen=true,       % Abre los marcadores

        bookmarkstype=toc,        % 
        pdfpagemode=UseOutlines,  %

    }
